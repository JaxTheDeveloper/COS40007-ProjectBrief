\subsection{Background \& Motivation}
\begin{frame}{1.1 Background \& Motivation}
    \frametitle{1.1 Background \& Motivation}
    \begin{itemize}
        \item \textbf{Core Motivation:} Upon team's familiarity with accelerator data (port. 2 \& 3).
        \item \textbf{Goal:} Extend scope to demonstrate real-world capabilities by constructing a Graphical User Interface (GUI) and from real accelerometer/gyroscope sensors.
        \item \textbf{Experience:} Team member has experience with IoT devices communicating sensor data and knowledge with embedded Linux systems (Jetson Orin Nano, Rasp. Pi, Arduino).
        \item \textbf{Expected Outcome:} Experiment with Data/AI workflow and understand Industrial AI applications.
    \end{itemize}
\end{frame}

\subsection{Target Users \& Tasks}
\begin{frame}
  \frametitle{1.2 Target Users \& Tasks}
  \begin{block}{[TBD] Stakeholders}

      \textbf{Supervisors:}
      \begin{itemize}
        \item Monitor worker activities and durations (e.g., boning, slicing).
        \item Evaluate equipment condition (sharpness).
        \item Grade performance and detect anomalies.
      \end{itemize}
        \textbf{Workers:}
      \begin{itemize}
        \item Effective way to clock in and clock out.
        \item Direct feedback on performance.
        \item Foster sense of responsibility and track work intensity.
      \end{itemize}

  \end{block}
\end{frame}

\subsection{Objectives}
\begin{frame}
  \frametitle{1.3 Objectives}
  \begin{block}{Primary Questions}
    \begin{enumerate}
      \item Can ML models accurately classify worker activities (cutting, slicing, idle) from motion sensors?
      \item Can we predict knife sharpness and determine optimal sharpening schedules?
    \end{enumerate}
  \end{block}
  
  \begin{itemize}
    \item \textbf{Goals, for now:} Develop a robust system with \textbf{\textgreater85\% classification accuracy}.
    \item \textbf{Benefits:}
    \begin{itemize}
        \item Enhanced safety (predictive maintenance).
        \item Reduced equipment downtime.
        \item Improved worker productivity.
        \item Data-driven decision making.
    \end{itemize}
  \end{itemize}
\end{frame}
